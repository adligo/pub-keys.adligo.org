# Open JDK 25

- [https://jdk.java.net/25/](https://jdk.java.net/25/)

This uses a environment variable FABRICATE_APPS_DIR which is usually /apps on Unix (Linux,Max) or /c/apps on Windows, but is configurable.

```
cd $FABRICATE_APPS_DIR

wget https://download.java.net/java/GA/jdk25.0.2/b1e0dfa218384cb9959bdcb897162d4e/10/GPL/openjdk-25.0.2_linux-x64_bin.tar.gz

# Check vs their sha256
wget https://download.java.net/java/GA/jdk25.0.2/b1e0dfa218384cb9959bdcb897162d4e/10/GPL/openjdk-25.0.2_linux-x64_bin.tar.gz.sha256
cat openjdk-25.0.2_linux-x64_bin.tar.gz.sha256
diff openjdk-25.0.2_linux-x64_bin.tar.gz.sha256 ~/pub-keys.adligo.org/pub-keys/net/java/download/java/GA/jdk25.0.2/b1e0dfa218384cb9959bdcb897162d4e/10/GPL/openjdk-25.0.2_linux-x64_bin.tar.gz.sha256

# also do your own 256 / 512
sha256sum openjdk-25.0.2_linux-x64_bin.tar.gz > openjdk-25.sha256
# compare our sha256 with theirs (java.nets), note ours has a file name and a new line
diff openjdk-25.sha256 openjdk-25.0.2_linux-x64_bin.tar.gz.sha256
# compare our sha256 with Adligo's
diff openjdk-25.sha256 ~/pub-keys.adligo.org/pub-keys/net/java/download/java/GA/jdk25.0.2/b1e0dfa218384cb9959bdcb897162d4e/10/GPL/openjdk-25.0.2_linux-x64_bin.tar.gz.sha256

sha512sum openjdk-25.0.2_linux-x64_bin.tar.gz > openjdk-25.sha512
cat openjdk-25.sha512
diff openjdk-25.sha512 ~/pub-keys.adligo.org/pub-keys/net/java/download/java/GA/jdk25.0.2/b1e0dfa218384cb9959bdcb897162d4e/10/GPL/openjdk-25.0.2_linux-x64_bin.tar.gz.sha512

tar -xvzf openjdk-25.0.2_linux-x64_bin.tar.gz

mkdir installs
mkdir installs/java
mv openjdk-25.* installs/java/
```
